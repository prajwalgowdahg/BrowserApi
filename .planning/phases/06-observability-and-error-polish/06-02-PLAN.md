export interface LogEntry {
  timestamp: string;   // ISO 8601
  sessionId: string;
  action: string;
  status: 'success' | 'fail';
  error?: string;
  durationMs?: number;
}

export class ActionLogService {
  append(sessionId: string, entry: Omit<LogEntry, 'timestamp' | 'sessionId'>): void;
  getLogs(sessionId: string): LogEntry[];
  clear(sessionId: string): void;
}

export const actionLogService: ActionLogService;

    - Before `return success(...)`: `actionLogService.append(sessionId, { action: '<action_name>', status: 'success', durationMs: Date.now() - startTime });`

    - Action names: 'navigate', 'click', 'type', 'select', 'screenshot_full', 'get_text', 'wait', 'scroll'

    **src/routes/compounds.ts** -- Add logging to all 4 compound handlers (login, fill_form, scrape, submit_form). Same try/catch pattern. Action names: 'login', 'fill_form', 'scrape', 'submit_form'. IMPORTANT: Log compound actions as a single action at the route handler level, NOT at the findElementWithAI sub-step level.

    **src/routes/sessions.ts** -- Add logging to:
    - POST / (create): After successful session creation, log with `action: 'session.create'`, status 'success'. Use `session.id` as the sessionId. Do NOT log on failure (capacity error returns via error() before this point).
